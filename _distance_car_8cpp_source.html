107 void DistanceCar::brake()
108 {
109  // Apply the opposite speed than the one currently applied to the motors
110  // until the car is moving slow enough and is practically stopped
111  auto currentSpeed = mCruiseControlEnabled ? mPreviousControlledSpeed : mTargetSpeed;
112  SimpleCar::setSpeed(-currentSpeed);
113  // Determine when we are moving slow enough to stop applying speed towards
114  // the opposite direction. We do not use the interrupt-based `getSpeed` from
115  // the odometers since when there are no pulses there will be no speed
116  // calculation.
117  for (auto attempt = 0; attempt < kMaxStopAttempts; attempt++)
118  {

120  mRuntime.delayMillis(kStopAttemptInterval);
121  auto distanceTravelled = getDistance() - initialDistance;
122  if (distanceTravelled == 0)
123  {
124  break;
125  }
126  }

129  mPreviousControlledSpeed = 0;

134 void DistanceCar::update()
135 {
136  if (!areOdometersAttached() || !mCruiseControlEnabled)
137  {
138  return;
139  }
140 
141  auto currentTime = mRuntime.currentTimeMillis();
142  if (currentTime < mPreviousUpdate + mFrequency)
143  {
144  return;
145  }
146  mPreviousUpdate = currentTime;
147 
148  // Don't try to immobilize the car if odometers are not directional since we
149  // have already done that when setting the speed
150  if (!areOdometersDirectional() && mTargetSpeed == kIdleControlSpeed)
151  {
152  return;
153  }
154  auto actualSpeed = getSpeed();
155  // If we do not have a way of determining direction of movement and the set
156  // speed is negative, then we assume that the car is also moving backwards.
157  // This assumption is based on the fact that if a speed towards the opposite
158  // direction is supplied we should make first sure we are stopped before
159  // changing direction.
160  if (!areOdometersDirectional() && mTargetSpeed < 0)
161  {
162  actualSpeed = -actualSpeed;
163  }
164  auto controlledSpeed = controlMotorSpeed(mPreviousControlledSpeed, mTargetSpeed, actualSpeed);
165  // In case direction readings are not supported by the odometers, we need to
166  // ensure that runaway will not occur (see #5) due to our assumption that we
167  // are moving at the same direction as our set speed.
168  if (!areOdometersDirectional() && (controlledSpeed * actualSpeed < 0))
169  {
170  // If we are unable to determine the direction of movement AND the PID controller output
171  // hints us to move towards the opposite direction, i.e. due to moving faster than we should
172  // or the odometers providing faulty readings, then don't try to spin the motors towards
173  // the opposite direction and just halt the motoros. This is because we do not have a
174  // reliable way to determine whether we are merely slowing down or accelerating towards
175  // the wrong (opposite) direction.
176  controlledSpeed = kIdleControlSpeed;
177  }
178  // Pass the rounded output of the pid as an input to the control
179  SimpleCar::setSpeed(lroundf(controlledSpeed));
180  // Save the unrounded float output of the PID controller
181  mPreviousControlledSpeed = controlledSpeed;
182 }
