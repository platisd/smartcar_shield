35 void GY50::update()
36 {
37  unsigned long currentTime = mRuntime.currentTimeMillis();
38  unsigned long interval = currentTime - mPreviousSample;
39  if (interval <= kSamplingInterval)
40  {
41  return; // Not the time to read yet
42  }
43 
44  static const float gyroSensitivity = 0.07f;
45  static const int gyroThreshold = 12; // Smaller changes are to be ignored
46  int drift = kOffset - getAngularVelocity();
47 
48  if (getAbsolute(drift) > gyroThreshold)
49  {
50  float gyroRate = drift * gyroSensitivity;
51  mAngularDisplacement += gyroRate / (1000.0 / interval);
52  }
53  mPreviousSample = currentTime;
54 }
