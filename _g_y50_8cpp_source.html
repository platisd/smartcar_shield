38 void GY50::update()
39 {
40  unsigned long currentTime = mRuntime.currentTimeMillis();
41  unsigned long interval = currentTime - mPreviousSample;
42  if (interval <= kSamplingInterval)
43  {
44  return; // Not the time to read yet
45  }
46 
47  int drift = kOffset - getAngularVelocity();
48 
49  if (getAbsolute(drift) > kGyroThreshold)
50  {
51  float gyroRate = static_cast<float>(drift) * kGyroSensitivity;
52  mAngularDisplacement += gyroRate / (1000.0f / static_cast<float>(interval));
53  }
54  mPreviousSample = currentTime;
55 }
